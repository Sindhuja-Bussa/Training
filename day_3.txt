#factorial

#include <stdio.h>
long long int factorial(long long int);
long long int n,z;
int main()
{
    scanf("%lld",&n);
    z=factorial(n);
    printf("%lld",z);
    return 0;
}
long long int factorial(long long int n)
{
    if(n==0 || n==1)
    return 1;
    else
    return n*factorial(n-1);
}

{

    printf("%d ",fib(i));
    }
    return 0;
}
int fib(int n)
{
    if(n==0 || n==1)
    return n;
    else
    return fib(n-1)+fib(n-2);
}
